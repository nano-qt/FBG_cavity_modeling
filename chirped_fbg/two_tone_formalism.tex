\section{Setup}

A CW laser at optical frequency $\omega_c$ is phase-modulated by an EOM at
frequency $\Delta$ with modulation index $m$. In the slowly-varying
envelope (carrier frame),
\begin{equation}
    E_{\mathrm{in}}(t) = E_0 \, e^{\,i m \sin(2\pi \Delta t)}
    = E_0 \sum_{k=-\infty}^{\infty} J_k(m)\, e^{\,i 2\pi k \Delta t}.
\end{equation}
For small $m$ we keep only $k \in \{-1,0,+1\}$ and use
$J_{-1}(m) = -J_1(m)$:
\begin{equation}
    E_{\mathrm{in}}(t) \approx E_0\Big[
    J_0(m) + J_1(m)\,e^{\,i 2\pi \Delta t}
    - J_1(m)\,e^{-i 2\pi \Delta t}
    \Big].
\end{equation}
The carrier is at detuning $\delta$ from the FBG resonance; the two
sidebands are at $\delta \pm \Delta$.

\section{Reflected field}

The FBG is linear and time-invariant, so each spectral component acquires
its own complex reflection coefficient:
\begin{equation}
    E_r(t) = E_0\Big[
    J_0\, r(\delta)
    + J_1\, r(\delta+\Delta)\, e^{\,i 2\pi \Delta t}
    - J_1\, r(\delta-\Delta)\, e^{-i 2\pi \Delta t}
    \Big],
\end{equation}
writing $J_k \equiv J_k(m)$ and leaving the common $e^{\,i\omega_c t}$
factor implicit.

\section{Photodetector output}

A square-law detector responds to $P(t) = |E_r(t)|^2$. Let
\begin{equation}
    A = J_0\, r(\delta), \qquad
    B = J_1\, r(\delta+\Delta), \qquad
    C = -J_1\, r(\delta-\Delta),
\end{equation}
so that $E_r / E_0 = A + B\, e^{i 2\pi\Delta t} + C\, e^{-i 2\pi\Delta t}$.
Multiplying out,
\begin{equation}
    \frac{P(t)}{|E_0|^2}
    = \underbrace{|A|^2 + |B|^2 + |C|^2}_{\text{DC}}
    + \underbrace{\big(A C^* + B A^*\big)\, e^{\,i 2\pi \Delta t}
                 + \mathrm{c.c.}}_{\text{beat at } \Delta}
    + \underbrace{B C^*\, e^{\,i 4\pi \Delta t} + \mathrm{c.c.}}_{\text{beat at } 2\Delta}.
\end{equation}
We keep only the component at $\Delta$.

\section{The $\Delta$ beat}

Substituting $A$, $B$, $C$,
\begin{equation}
    A C^* + B A^*
    = -J_0 J_1\, r(\delta)\, r^*(\delta-\Delta)
      + J_0 J_1\, r(\delta+\Delta)\, r^*(\delta).
\end{equation}
Define the (normalized) complex observable
\begin{equation}
    \boxed{\;
    \mathcal{C}(\delta;\Delta)
    \;\equiv\; r^*(\delta)\,r(\delta+\Delta) \;-\; r(\delta)\,r^*(\delta-\Delta)
    \;}
\end{equation}
so that
\begin{equation}
    P(t) = P_{\mathrm{DC}} + 2\,|E_0|^2 J_0 J_1 \,
    \Re\!\big[\mathcal{C}(\delta;\Delta)\, e^{\,i 2\pi \Delta t}\big]
    + (\text{terms at } 2\Delta).
\end{equation}

\section{Demodulation}

Mixing $P(t)$ with an electrical LO at $\Delta$ and low-passing selects
one quadrature of $\mathcal{C}$:
\begin{equation}
    \big\langle P(t)\, e^{-i 2\pi \Delta t}\big\rangle_t
    = |E_0|^2 J_0(m) J_1(m)\,\mathcal{C}(\delta;\Delta).
\end{equation}
Equivalently, a Fourier transform of $P(t)$ evaluated at $+\Delta$ yields
the same $\mathcal{C}$ up to window factors.

\section{Experimental observable}

The directly measured power at frequency $\Delta$ is
\begin{equation}
    |\mathcal{C}(\delta;\Delta)|^2
    = \big|\,r^*(\delta)\,r(\delta+\Delta) - r(\delta)\,r^*(\delta-\Delta)\,\big|^2.
\end{equation}
The phase $\arg\mathcal{C}$ is obtained by I/Q demodulation.

\section{Small-$\Delta$ expansion}

Write $r(\delta) = R(\delta)\, e^{\,i\phi(\delta)}$ and Taylor expand
$r(\delta\pm\Delta) = r \pm \Delta\, r' + \tfrac{1}{2}\Delta^2\, r'' + O(\Delta^3)$.
Taking the conjugate on the $-\Delta$ side (conjugation and
$d/d\delta$ commute for real $\delta$),
\begin{align}
    r^*(\delta)\,r(\delta+\Delta)
    &= |r|^2 + \Delta\, r^* r' + \tfrac{1}{2}\Delta^2\, r^* r'' + O(\Delta^3), \\
    r(\delta)\,r^*(\delta-\Delta)
    &= |r|^2 - \Delta\, r\, r'^{\,*} + \tfrac{1}{2}\Delta^2\, r\, r''^{\,*} + O(\Delta^3).
\end{align}
Subtracting,
\begin{equation}
    \mathcal{C}(\delta;\Delta)
    = \Delta\,\big(r^* r' + r\, r'^{\,*}\big)
    + \tfrac{1}{2}\Delta^2\,\big(r^* r'' - r\, r''^{\,*}\big)
    + O(\Delta^3).
\end{equation}

Using $r^* r' = \tfrac{1}{2}\,d|r|^2/d\delta + i\,|r|^2 \phi'$, the first
bracket is purely real and the second purely imaginary:
\begin{equation}
    r^* r' + r\, r'^{\,*} = 2\,\Re(r^* r') = \frac{d|r|^2}{d\delta}, \qquad
    r^* r'' - r\, r''^{\,*} = 2i\,\Im(r^* r'').
\end{equation}
A short calculation with $r = R e^{i\phi}$ gives
$\Im(r^* r'') = 2 R R' \phi' + R^2 \phi'' = d(R^2 \phi')/d\delta$. Hence
\begin{equation}
\boxed{\;
    \mathcal{C}(\delta;\Delta)
    \;\approx\; \Delta\,\frac{d|r|^2}{d\delta}
    \;+\; i\,\Delta^2\,\frac{d\!\left(|r|^2 \phi'\right)}{d\delta}
    \;+\; O(\Delta^3).
\;}
\end{equation}

The parity $\mathcal{C}(-\Delta) = -\mathcal{C}^*(\Delta)$ forces odd
coefficients to be real and even coefficients imaginary; the expansion
above satisfies this.

\paragraph{Consequences.}
\begin{itemize}\setlength\itemsep{2pt}
  \item $\Re\mathcal{C}$ is \emph{leading order} in $\Delta$ and tracks
        $d|r|^2/d\nu$ — the amplitude derivative. Classical PDH
        error-signal shape.
  \item $\Im\mathcal{C}$ is \emph{one order smaller} and mixes $\phi'$ with
        the amplitude variation $R'$. It is not a clean proxy for
        $d\phi/d\nu$ unless $R$ is approximately constant over the scan.
\end{itemize}

\section{Phase-slope reconstruction}

The cleanest finite-$\Delta$ estimator of $d\phi/d\nu$ uses the raw phases
at the two sideband frequencies directly (both accessible once $r$ is
sampled, and experimentally via I/Q demodulation giving complex
amplitudes rather than power):
\begin{equation}
\boxed{\;
    \widehat{\phi'}(\delta)
    \;=\; \frac{\arg r(\delta+\Delta) - \arg r(\delta-\Delta)}{2\,\Delta}
    \;=\; \phi'(\delta) + \tfrac{1}{6}\Delta^2\,\phi'''(\delta) + O(\Delta^4).
\;}
\end{equation}
Any branch jumps in $\arg r$ must be unwrapped before differencing.

Using $\mathcal{C}$ alone is an option only in the regime
$R'\!/R \ll \phi'$, where
$\Im\mathcal{C}/(\Delta^2\,|r|^2) \approx \phi'(\delta) + \phi'\,R'/R \to \phi'$.

\section{Implementation notes}

\begin{itemize}\setlength\itemsep{2pt}
  \item $r$ is stored on a detuning grid; off-grid $r(\delta\pm\Delta)$
        is evaluated by complex linear interpolation (real and imaginary
        parts independently).
  \item Scan $\delta \in [\delta_\min + \Delta,\;\delta_\max - \Delta]$ to
        keep both sidebands in-band; outside this range truncate.
  \item The overall prefactor $|E_0|^2 J_0(m) J_1(m)$ is absorbed into
        normalization; only the $\delta$- and $\Delta$-dependence of
        $\mathcal{C}$ matters.
\end{itemize}
